	vowels - ['a','e','i','o','u'].

	temp = []
	for i in range(len(string)):
    		if string[i]=='a' or string[i]=='e' or string[i]=='i' or string[i]=='o' or string[i]=='u':
        		if(string[i] not in temp):
            			temp.append(string[i])
